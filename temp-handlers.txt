        method: 'POST',
        body: formData,
        credentials: 'include'
      })
      
      if (!response.ok) {
        const errorText = await response.text()
        throw new Error(`Gagal mengimpor data (${response.status}): ${errorText}`)
      }
      
      const result = await response.json()
      
      if (result.success) {
        toast({
          title: "Berhasil",
          description: `${result.imported || 0} data unit kerja berhasil diimpor`,
          variant: "default",
